import pandas as pd
import numpy as np
from sklearn.preprocessing import MinMaxScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_absolute_error, r2_score
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import GRU, Dense, Dropout
from tensorflow.keras.callbacks import EarlyStopping

# Define constants
SEQUENCE_LENGTH = 30

# Pipeline Function
def preprocess_data(data, sequence_length=SEQUENCE_LENGTH, is_test=False):
    """Preprocess the data: normalize, create sequences, and handle missing values."""
    # Normalize
    scaler = MinMaxScaler()
    data.iloc[:, 2:] = scaler.fit_transform(data.iloc[:, 2:])
    
    # Generate sequences
    def create_sequences(data, sequence_length=30, drop_rul=True):
        sequences, labels = [], []
        for unit in data['Unit'].unique():
            unit_data = data[data['Unit'] == unit].sort_values('Time')
            for i in range(len(unit_data) - sequence_length):
                sequence = unit_data.iloc[i:i+sequence_length].drop(columns=['Unit', 'Time']).values
                sequences.append(sequence)
                if drop_rul and 'RUL' in unit_data.columns:
                    labels.append(unit_data.iloc[i + sequence_length - 1]['RUL'])
        if drop_rul and 'RUL' in data.columns:
            return np.array(sequences), np.array(labels)
        return np.array(sequences)

    if is_test:
        return create_sequences(data, sequence_length, drop_rul=False)
    return create_sequences(data, sequence_length)

# Define GRU Model
def build_gru_model(input_shape):
    model = Sequential()
    model.add(GRU(128, input_shape=input_shape, return_sequences=False))
    model.add(Dropout(0.2))
    model.add(Dense(64, activation='relu'))
    model.add(Dropout(0.2))
    model.add(Dense(1))
    model.compile(optimizer='adam', loss='mean_squared_error', metrics=['mae', 'mse'])
    return model

# Train GRU Model
gru_model = build_gru_model((SEQUENCE_LENGTH, X_train_gru.shape[2]))
early_stopping = EarlyStopping(monitor='val_loss', patience=5, restore_best_weights=True)

history = gru_model.fit(
    X_train_gru, y_train_gru,
    validation_data=(X_val_gru, y_val_gru),
    epochs=50, batch_size=32,
    callbacks=[early_stopping]
)

# Evaluate on Validation Data
